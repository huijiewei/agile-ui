---
title: 头像
description: 用作头像显示，可以为图片、图标或字符形式展示。
---

## 导入

```tsx
import { Avatar, AvatarBadge, AvatarGroup } from '@agile-ui/react';
```

## 使用

```tsx preview editable
<div className={'flex flex-row gap-3'}>
  <Avatar src={'https://yuncars-other.oss-cn-shanghai.aliyuncs.com/boilerplate/avatar/a00430.png@!avatar'} />

</div>
```
