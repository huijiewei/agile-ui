---
title: 分页
description: 采用分页控制单页内的信息数量，也可进行页面跳转。
---

## 导入

```tsx
import { Pagination } from '@agile-ui/react';
```

## 使用

```tsx preview editable
<Pagination
  onChange={(page) => {
    console.log(page);
  }}
  total={2120}
/>
```

## 禁用状态

```tsx preview editable
<Pagination
  disabled
  onChange={(page) => {
    console.log(page);
  }}
  total={2120}
/>
```
