---
title: 弹出框
description: 鼠标点击在某个组件时，弹出的气泡式的卡片浮层。可以对卡片上的元素进行操作。
---

## 用法

## 带有箭头

```tsx preview editable
<Popover>
  <PopoverTrigger>
    <Button>开启</Button>
  </PopoverTrigger>
  <PopoverContent>
    <PopoverArrow />
    <PopoverHeader>带有箭头弹出框</PopoverHeader>
    <PopoverBody>这是一个带有箭头弹出框</PopoverBody>
  </PopoverContent>
</Popover>
```

## 回调事件

```tsx preview editable
<Popover onClose={() => console.log('弹出框关闭回调')} placement={'right'}>
  <PopoverTrigger>
    <Button>开启</Button>
  </PopoverTrigger>
  <PopoverContent>
    <PopoverHeader>回调弹出框</PopoverHeader>
    <PopoverBody>这是一个回调弹出框</PopoverBody>
  </PopoverContent>
</Popover>
```

## 可控

```tsx preview editable
function Demo() {
  const [opened, setOpened] = React.useState(false);

  return (
    <Popover opened={opened} placement={'right'}>
      <PopoverTrigger>
        <Button onMouseEnter={() => setOpened(true)} onMouseLeave={() => setOpened(false)}>
          开启
        </Button>
      </PopoverTrigger>
      <PopoverContent>
        <PopoverHeader>回调弹出框</PopoverHeader>
        <PopoverBody>这是一个回调弹出框</PopoverBody>
      </PopoverContent>
    </Popover>
  );
}
```

## 表单

```tsx preview editable
<Popover closeOnBlur={false} placement={'right'}>
  {({ opened, handleClose }) => (
    <>
      <PopoverTrigger>
        <Button>{opened ? '关闭' : '开启'}表单</Button>
      </PopoverTrigger>
      <PopoverContent>
        <PopoverHeader>弹出框表单</PopoverHeader>
        <PopoverBody>
          <FormDemo />
        </PopoverBody>
        <PopoverFooter className={'flex items-center justify-center gap-5'}>
          <Button>确定</Button>
          <Button color={'gray'} onClick={handleClose}>
            关闭
          </Button>
        </PopoverFooter>
      </PopoverContent>
    </>
  )}
</Popover>
```

## 放置位置

```tsx preview editable
<div
  className={'grid gap-2 mx-auto w-fit'}
  style={{
    gridTemplateAreas:
      '".           top-start     top         top-end       .            "' +
      '"left-start  .             .           .             right-start  "' +
      '"left        .             center      .             right        "' +
      '"left-end    .             .           .             right-end    "' +
      '".           bottom-start  bottom      bottom-end    .            "',
  }}
>
  {[
    ['top-start', '上左'],
    ['top', '上'],
    ['top-end', '上右'],
    ['left-start', '左上'],
    ['left', '左'],
    ['left-end', '左下'],
    ['auto', '自动'],
    ['right-start', '右上'],
    ['right', '右'],
    ['right-end', '右下'],
    ['bottom-start', '下左'],
    ['bottom', '下'],
    ['bottom-end', '下右'],
  ].map(([placement, name]) => (
    <Popover key={placement} placement={placement}>
      <PopoverTrigger>
        <button
          style={{ gridArea: placement == 'auto' ? 'center' : placement }}
          className={
            'flex justify-center w-12 h-12 items-center bg-gray-200 rounded cursor-default hover:bg-gray-100 active:bg-gray-300'
          }
        >
          {name}
        </button>
      </PopoverTrigger>
      <PopoverContent>
        <PopoverArrow />
        <PopoverCloseButton />
        <PopoverHeader>标题</PopoverHeader>
        <PopoverBody>这是一个在{name}位置的弹出框</PopoverBody>
        <PopoverFooter>底部显示</PopoverFooter>
      </PopoverContent>
    </Popover>
  ))}
</div>
```
