## 可控

```tsx preview editable
function Example() {
  const [value, setValue] = React.useState('');
  const handleChange = (value) => setValue(value);

  return (
    <>
      <div className="mb-2">值: {value}</div>
      <Input value={value} onChange={handleChange} placeholder="输入框" />
    </>
  );
}
```
