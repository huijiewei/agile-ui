---
title: 下拉菜单
description: 页面上的命令过多时，可将备选命令收纳到向下展开的浮层容器中。
---

## 用法

```tsx preview editable
<DropdownMenu>
  <DropdownMenuTrigger as={Button}>下拉菜单</DropdownMenuTrigger>
  <DropdownMenuContent>
    <DropdownMenuArrow />
    <DropdownMenuItem
      onClick={() => {
        alert('点击了新标签页');
      }}
    >
      新标签页<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⌘+T</span>
    </DropdownMenuItem>
    <DropdownMenuItem>
      打开新窗口<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⌘+N</span>
    </DropdownMenuItem>
    <DropdownMenuItem disabled>
      打开新的无痕式窗口<span className={'ml-auto pl-5 text-sm'}>⇧+⌘+N</span>
    </DropdownMenuItem>
    <DropdownMenuDivider />
    <DropdownMenuItem>
      关闭窗口<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⇧+⌘+W</span>
    </DropdownMenuItem>
    <DropdownMenuItem>
      关闭标签页<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⌘+W</span>
    </DropdownMenuItem>
    <DropdownMenuItem>
      页面存储为...<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⌘+S</span>
    </DropdownMenuItem>
    <DropdownMenuDivider />
    <DropdownMenu>
      <DropdownMenuItemTrigger>分享</DropdownMenuItemTrigger>
      <DropdownMenuContent>
        <DropdownMenuItem
          onClick={() => {
            alert('点击了电子邮件');
          }}
        >
          电子邮件
        </DropdownMenuItem>
        <DropdownMenuItem>信息</DropdownMenuItem>
        <DropdownMenuItem>隔空投送</DropdownMenuItem>
      </DropdownMenuContent>
    </DropdownMenu>
    <DropdownMenuDivider />
    <DropdownMenuItem>
      打印<span className={'ml-auto text-gray-500 pl-5 text-sm'}>⌘+P</span>
    </DropdownMenuItem>
  </DropdownMenuContent>
</DropdownMenu>
```
