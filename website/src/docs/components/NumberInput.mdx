
## 格式化

```tsx preview editable
<div className={'flex flex-row items-center gap-3'}>
  <NumberInput
    parse={(value) => value.replace(/\$\s?|(,*)/g, '')}
    format={(value) => (!Number.isNaN(value) ? `$ ${value}`.replace(/\B(?=(\d{3})+(?!\d))/g, ',') : '$ ')}
  />
</div>
```
