## 导入

```tsx
import { SpinnerOverlay } from '@agile-ui/react';
```

## 用法

```tsx live editable
function SpinnerOverlayDemo() {
  const [visible, setVisible] = React.useState(false);

  return (
    <>
      <div className={'relative w-fit mx-auto p-5 shadow-md'}>
        <SpinnerOverlay visible={visible} />
        <FormDemo />
      </div>

      <div className={'flex mt-3 justify-center'}>
        <Button onClick={() => setVisible((v) => !v)}>切换覆盖层</Button>
      </div>
    </>
  );
}
```

## 自定义加载器

```tsx live editable
<div className={'relative w-fit p-5 mx-auto shadow-md'}>
  <SpinnerOverlay visible={true}>
    <Spinner color={'green'} size={'xl'} />
  </SpinnerOverlay>
  <FormDemo />
</div>
```
